常见问题解决

  • 音效不工作？ - 这是正常现象！由于浏览器限制，音效需在用户第一次交互后才能激活
  • 尝试点击页面上的按钮 - 这可以"唤醒"音频系统
  • 调整音量 - 确保音量没有设置为0
  • 切换音效类型 - 部分音效声音可能很小

实现原理

// 点击处理函数
function handleClick(e) {
  // 检查是否是可点击元素
  if (!isClickable(e.target)) return;
  
  // 处理浏览器自动播放限制
  if (!audioInitialized) {
    initAudioSystem(); // 第一次点击时初始化音频系统
    return;
  }
  
  // 创建涟漪效果
  createRipple(e);
  
  // 播放音效
  playClickSound();
}

// 初始化音频系统
function initAudioSystem() {
  try {
    audioContext = new (window.AudioContext || window.webkitAudioContext)();
    loadSounds(); // 加载所有音效
    audioInitialized = true;
    updateStatus("音效系统已激活", "success");
  } catch (e) {
    updateStatus("浏览器不支持Web Audio API", "error");
  }
}

// 播放音效
function playClickSound() {
  if (!audioContext || !audioInitialized) return;
  
  const source = audioContext.createBufferSource();
  source.buffer = audioBuffers[currentSoundType];
  const gainNode = audioContext.createGain();
  gainNode.gain.value = volume;
  source.connect(gainNode);
  gainNode.connect(audioContext.destination);
  source.start(0);
}
注意：某些浏览器在用户第一次与页面交互前会阻止音效播放
